import javafx.application.Application;
import javafx.event.ActionEvent;
import javafx.event.EventHandler;
import javafx.geometry.Insets;
import javafx.scene.AccessibleAction;
import javafx.scene.Scene;
import javafx.scene.control.*;
import javafx.scene.layout.HBox;
import javafx.scene.layout.VBox;
import javafx.scene.text.Font;
import javafx.stage.FileChooser;
import javafx.stage.Stage;
import javafx.geometry.Pos;

import java.io.File;
import java.io.FileNotFoundException;
import java.io.PrintWriter;

public class App extends Application{
    private Button btn_calculate , btn_export;
    private Label lbl_initialBalanceTitle ,
            lbl_interestRateTitle,lbl_years, lbl_yearsHint, lbl_cye,
    lbl_cyeHint ,lbl_totalBalance , lbl_totalBalanceHint;
    private TextField tf_initialBalance, tf_interestRate;
    private TextArea ta_history;
    private double balance;
    private int years;
    public static void main(String[] args) {
        launch(args);
    }

    @Override
    public void start(Stage stage) throws Exception {
        initialize();
        // Line 1
        HBox inputPane = new HBox(10,lbl_initialBalanceTitle,
                tf_initialBalance, lbl_interestRateTitle,
                tf_interestRate);

        //Line 2 btn_calculate
        //Line 3
        VBox yearsPane = new VBox(10,lbl_years,lbl_yearsHint);
        VBox cyePane = new VBox(10,lbl_cye,lbl_cyeHint);
        VBox totalBalancePane = new VBox(10,lbl_totalBalance,
                lbl_totalBalanceHint);
        HBox outputPane = new HBox(10,yearsPane,cyePane,totalBalancePane);
        //Line 4 ta_history
        //Line 5 btn_export
        //rootPane
        VBox rootPane = new VBox(10,inputPane, btn_calculate,
                outputPane,ta_history,btn_export);
        lbl_initialBalanceTitle.setPadding(new Insets(8));
        inputPane.setAlignment(Pos.CENTER);
        yearsPane.setAlignment(Pos.CENTER);
        cyePane.setAlignment(Pos.CENTER);
        totalBalancePane.setAlignment(Pos.CENTER);
        outputPane.setAlignment(Pos.CENTER);
        rootPane.setAlignment(Pos.CENTER);
        btn_calculate.setMaxWidth(Double.MAX_VALUE);
        btn_export.setMaxWidth(Double.MAX_VALUE);


        rootPane.setMinSize(500,500);
        lbl_initialBalanceTitle.setFont(new Font("Arial",40));
        setPadding(8,lbl_initialBalanceTitle ,
                lbl_interestRateTitle,lbl_years, lbl_yearsHint, lbl_cye,
                lbl_cyeHint ,lbl_totalBalance , lbl_totalBalanceHint,
                btn_calculate , btn_export,tf_initialBalance,
                tf_interestRate,ta_history);
        setFontSize(16,lbl_initialBalanceTitle ,
                lbl_interestRateTitle,lbl_years, lbl_yearsHint, lbl_cye,
                lbl_cyeHint ,lbl_totalBalance , lbl_totalBalanceHint,
                btn_calculate , btn_export,tf_initialBalance,
                tf_interestRate,ta_history);
        setFontSize(40,lbl_years,lbl_cye,lbl_totalBalance);
        lbl_years.setStyle("-fx-font-weight:bold;-fx-text-fill:red");
        lbl_yearsHint.setStyle("-fx-text-fill:red");
        lbl_cye.setStyle("-fx-font-weight:bold;-fx-text-fill:red");
        lbl_cyeHint.setStyle("-fx-text-fill:red");
        lbl_totalBalance.setStyle("-fx-font-weight:bold;-fx-text-fill:red");
        lbl_totalBalanceHint.setStyle("-fx-text-fill:red");

        Scene scene = new Scene(rootPane);
        stage.setScene(scene);
        stage.setTitle("Investment App");
        btn_calculate.setOnAction(new EventHandler<ActionEvent>() {
            @Override
            public void handle(ActionEvent actionEvent) {
               double initialBalance = Double.parseDouble(tf_initialBalance.getText().trim());
               double intrestRate = Double.parseDouble(tf_interestRate.getText().trim());
               if (years ==0) {
                   balance = initialBalance;//for the first year
               }
               years++;
               double cyi = balance * intrestRate /100;//ربح السنة
               balance+=cyi;//الربح الكلي
                lbl_years.setText(String.valueOf(years));
                lbl_cye.setText(String.format("$%,.2f",cyi));
                lbl_totalBalance.setText(String.format("$%,.2f",balance));
                ta_history.appendText("Years #: "+years+
                        " You earned: "+ String.format("$%,.2f",cyi)+
                        " Your total balance is: "+String.format("$%,.2f",balance)+"\n");

            }

        });
        btn_export.setOnAction(new EventHandler<ActionEvent>() {
            @Override
            public void handle(ActionEvent actionEvent) {
                FileChooser fc = new FileChooser();
                fc.setTitle("Save Investment File");
                File f = fc.showSaveDialog(stage);
                try(PrintWriter pw = new PrintWriter(f)
                ){
                    pw.println(ta_history.getText());
                }
                catch (FileNotFoundException e){

                }

            }
        });

        stage.show();
    }
    private void initialize(){
        btn_calculate = new Button("Calculate");
        btn_export = new Button("Export");
        lbl_initialBalanceTitle = new Label("Initial Balance");
        lbl_interestRateTitle = new Label("Interest Rate");
        lbl_years = new Label("0");
        lbl_yearsHint = new Label("Yeras");
        lbl_cye = new Label("0");
        lbl_cyeHint = new Label("Current Year Eearning");
        lbl_totalBalance = new Label("0");
        lbl_totalBalanceHint = new Label("Total Balance");
        tf_initialBalance= new TextField();
        tf_interestRate = new TextField();
        ta_history = new TextArea();
    }
    private void setPadding(double padding , Control ... controls){
        for (Control c : controls){
            c.setPadding(new Insets(padding));
        }


    }
    private void setFontSize(double fontSize , Control ... controls){
        for (Control c : controls){
          if(c instanceof TextInputControl){
              ((TextInputControl)c).setFont(new Font(fontSize));
          }
          else if(c instanceof Labeled){
              ((Labeled) c).setFont(new Font(fontSize));
          }
        }
    }
}
